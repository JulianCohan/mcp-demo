Building a GitHub MCP Server

github_server.py 
# MCP Server for GitHub Integration
import mcp
from mcp.server import Server
from github import Github

server = Server("github-mcp")
github_client = Github(token=os.getenv("GITHUB_TOKEN"))

@server.list_resources()
async def list_resources():
    """List available GitHub resources"""
    return [
        mcp.Resource(
            uri="github://repos",
            name="GitHub Repositories",
            description="Access repository information"
        ),
        mcp.Resource(
            uri="github://issues",
            name="GitHub Issues", 
            description="Repository issues and PRs"
        )
    ]

@server.read_resource()
async def read_resource(uri: str):
    """Read GitHub resource data"""
    if uri.startswith("github://repos/"):
        repo_name = uri.split("/")[-1]
        repo = github_client.get_repo(repo_name)
        return {
            "name": repo.name,
            "stars": repo.stargazers_count,
            "forks": repo.forks_count,
            "description": repo.description,
            "language": repo.language,
            "last_updated": repo.updated_at.isoformat()
        }

@server.list_tools()
async def list_tools():
    """List available GitHub tools"""
    return [
        mcp.Tool(
            name="create_issue",
            description="Create a new GitHub issue",
            input_schema={
                "type": "object",
                "properties": {
                    "repo": {"type": "string"},
                    "title": {"type": "string"},
                    "body": {"type": "string"}
                },
                "required": ["repo", "title"]
            }
        )
    ]

@server.call_tool()
async def call_tool(name: str, arguments: dict):
    """Execute GitHub tool"""
    if name == "create_issue":
        repo = github_client.get_repo(arguments["repo"])
        issue = repo.create_issue(
            title=arguments["title"],
            body=arguments.get("body", "")
        )
        return {
            "success": True,
            "issue_number": issue.number,
            "url": issue.html_url
        }
ai_app.py 
# AI Application using GitHub MCP Server
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# Connect to GitHub MCP server
server_params = StdioServerParameters(
    command="python",
    args=["github_server.py"]
)

async def analyze_repo_with_ai():
    async with stdio_client(server_params) as session:
        # Discover available resources
        resources = await session.list_resources()
        print(f"Available resources: {[r.name for r in resources]}")
        
        # Read repository data
        repo_data = await session.read_resource("github://repos/anthropic/claude")
        
        # AI can now analyze this data
        analysis_prompt = f"""
        Analyze this repository:
        Name: {repo_data['name']}
        Stars: {repo_data['stars']}
        Language: {repo_data['language']}
        Description: {repo_data['description']}
        
        Provide insights about its popularity and technology stack.
        """
        
        # Use available tools
        tools = await session.list_tools()
        if "create_issue" in [tool.name for tool in tools]:
            # AI decides to create a follow-up issue
            result = await session.call_tool("create_issue", {
                "repo": "anthropic/claude",
                "title": "AI Analysis: Repository Insights",
                "body": "Automated analysis suggests improvements..."
            })
            print(f"Created issue: {result['url']}")

# The magic: AI can now interact with GitHub
# through a standardized protocol, regardless of
# the underlying implementation details

  }
